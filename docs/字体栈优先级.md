<h1 id="adobe_fonts">全局思源黑体</h1>

<h2 id="adobe_fonts_uni">通用</h2>

```css
/* 无衬线字体 */
font-family: "思源黑体",
             "Noto Sans CJK SC",
             -apple-system,
             BlinkMacSystemFont,
             system-ui,
             "Apple Color Emoji",
             "Segoe UI Emoji",
             "Noto Color Emoji",
             sans-serif;

/* 等宽字体 */
font-family: "思源等宽",
             "Noto Sans Mono CJK SC",
             "Source Code Pro",
             "思源黑体",
             "Noto Sans CJK SC",
             "Apple Color Emoji",
             "Segoe UI Emoji",
             "Noto Color Emoji",
             monospace;
```

<h2 id="adobe_fonts_hc">繁体中文 (香港)、繁体中文 (澳门)</h2>

```css
/* 无衬线字体 */
/* Google 目前并未推出“Noto Sans CJK TC”，此处仅为提前适配 */
font-family: "思源黑體 香港"
             "Noto Sans CJK HC",
             "思源黑體",
             "Noto Sans CJK TC",
             "HanHei TC",
             -apple-system,
             BlinkMacSystemFont,
             system-ui,
             "Apple Color Emoji",
             "Segoe UI Emoji",
             "Noto Color Emoji",
             sans-serif;

/* 衬线字体 */
/* Adobe 目前并未推出“思源宋體 香港”，Google 也未推出“Noto Serif CJK HC”，此处仅为提前适配 */
font-family: "思源宋體 香港"
             "Noto Serif CJK HC",
             "思源宋體",
             "Noto Serif CJK TC",
             "Source Serif Pro",
             "Apple Color Emoji",
             "Segoe UI Emoji",
             "Noto Color Emoji",
             serif;

/* 等宽字体 */
/* Google 目前并未推出“Noto Sans Mono CJK HC”，此处仅为提前适配 */
font-family: "思源等寬 香港",
             "Noto Sans Mono CJK HC",
             "思源等寬",
             "Noto Sans Mono CJK TC",
             "Source Code Pro",
             "思源黑體 香港"
             "Noto Sans CJK HC",
             "思源黑體",
             "Noto Sans CJK TC",
             "HanHei TC",
             "Apple Color Emoji",
             "Segoe UI Emoji",
             "Noto Color Emoji",
             monospace;
```
